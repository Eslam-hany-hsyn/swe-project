CREATE OR REPLACE PROCEDURE Filter_Results_Events ( p_title VARCHAR2, p_startDate DATE, p_endDate DATE, p_record OUT SYS_REFCURSOR ) 
AS
BEGIN
    OPEN p_record FOR
        SELECT  E.TITLE, E.DESCRIPTION, E.STATUS, TS.SLOTDATE, TS.STARTTIME, TS.ENDTIME
        FROM EVENTS E
        INNER JOIN TIMESLOTS TS ON E.TIMESLOTID = TS.TIMESLOTID
        WHERE E.STATUS = 'Approved' 
        AND (E.TITLE LIKE '%' || p_title || '%' OR p_title IS NULL)
        AND TS.SLOTDATE BETWEEN p_startDate AND p_endDate;
END Filter_Results_Events;

CREATE OR REPLACE PROCEDURE Register_Or_Cancel_Event ( p_eventID NUMBER, p_attendeeID NUMBER, p_status VARCHAR2 )
AS
BEGIN
    IF p_status = 'register' THEN
        INSERT INTO EVENTREGISTRCANCEL (EVENTID, ATTENDEEID)
        VALUES (p_eventID, p_attendeeID);
    ELSIF p_status = 'cancel' THEN
        DELETE FROM EVENTREGISTRCANCEL 
        WHERE EVENTID = p_eventID AND ATTENDEEID = p_attendeeID;
    END IF;
    
    COMMIT;
END Register_Or_Cancel_Event ;

CREATE OR REPLACE PROCEDURE Get_All_Approved_Events ( p_record OUT SYS_REFCURSOR ) 
AS
BEGIN
    OPEN p_record FOR
        SELECT  E.EVENTID, E.TITLE, E.DESCRIPTION, TS.SLOTDATE, TS.STARTTIME, TS.ENDTIME, E.CAPACITY, E.STATUS
        FROM EVENTS E
        INNER JOIN TIMESLOTS TS ON E.TIMESLOTID = TS.TIMESLOTID
        WHERE E.STATUS = 'Approved'
        ORDER BY TS.SLOTDATE ASC; 
END;
