CREATE OR REPLACE PROCEDURE Get_All_Approved_Events ( p_record OUT SYS_REFCURSOR ) 
AS
BEGIN
    OPEN p_record FOR
        SELECT  E.EVENTID, E.TITLE, CAST(E.DESCRIPTION AS VARCHAR2(4000)) AS DESCRIPTION, TS.SLOTDATE, TS.STARTTIME, TS.ENDTIME, E.CAPACITY, E.STATUS
        FROM EVENTS E
        JOIN TIMESLOTS TS ON E.TIMESLOTID = TS.TIMESLOTID
        WHERE E.STATUS = 'Approved'
        ORDER BY TS.SLOTDATE ASC;
END;

create or replace PROCEDURE Filter_Results_Events ( p_title VARCHAR2, p_record OUT SYS_REFCURSOR ) 
AS
BEGIN
    OPEN p_record FOR
        SELECT  E.EVENTID, E.TITLE, CAST(E.DESCRIPTION AS VARCHAR2(4000)) AS DESCRIPTION, TS.SLOTDATE, TS.STARTTIME, TS.ENDTIME, E.CAPACITY, E.STATUS
        FROM EVENTS E
        INNER JOIN TIMESLOTS TS ON E.TIMESLOTID = TS.TIMESLOTID
        WHERE (p_title IS NULL OR LOWER(E.TITLE) LIKE '%' || LOWER(p_title) || '%')
        AND E.STATUS = 'Approved';
END Filter_Results_Events;

create or replace PROCEDURE Filter_Results_Date_Events ( p_date DATE, p_record OUT SYS_REFCURSOR ) 
AS
BEGIN
    OPEN p_record FOR
        SELECT  E.EVENTID, E.TITLE, CAST(E.DESCRIPTION AS VARCHAR2(4000)) AS DESCRIPTION, TS.SLOTDATE, TS.STARTTIME, TS.ENDTIME, E.CAPACITY, E.STATUS
        FROM EVENTS E
        INNER JOIN TIMESLOTS TS ON E.TIMESLOTID = TS.TIMESLOTID
        WHERE (TS.SLOTDATE = p_date)
        AND E.STATUS = 'Approved';
END Filter_Results_Date_Events;

CREATE OR REPLACE PROCEDURE Register_Or_Cancel_Event ( p_eventID NUMBER, p_attendeeID NUMBER, p_status VARCHAR2 )
AS
BEGIN
    IF p_status = 'register' THEN
        INSERT INTO EVENTREGISTRCANCEL (EVENTID, ATTENDEEID)
        VALUES (p_eventID, p_attendeeID);
    ELSIF p_status = 'cancel' THEN
        DELETE FROM EVENTREGISTRCANCEL 
        WHERE EVENTID = p_eventID AND ATTENDEEID = p_attendeeID;
    END IF;
    
    COMMIT;
END Register_Or_Cancel_Event ;
