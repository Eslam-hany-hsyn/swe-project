-- =========================
-- ACCOUNTS
-- =========================
CREATE TABLE Accounts (
    AccountID NUMBER,
    UserName NVARCHAR2(50) NOT NULL,
    Password NVARCHAR2(255) NOT NULL,
    CONSTRAINT PK_Accounts PRIMARY KEY (AccountID)
);

CREATE SEQUENCE seq_accounts START WITH 1;

CREATE OR REPLACE TRIGGER trg_accounts
BEFORE INSERT ON Accounts
FOR EACH ROW
WHEN (NEW.AccountID IS NULL)
BEGIN
    :NEW.AccountID := seq_accounts.NEXTVAL;
END;
/

-- =========================
-- PERSONS
-- =========================
CREATE TABLE Persons (
    PersonID NUMBER,
    AccountID NUMBER,
    Name NVARCHAR2(100) NOT NULL,
    Phone NVARCHAR2(20),

-- =========================
-- EVENTS
-- =========================
CREATE TABLE Events (
    EventID NUMBER,
    OrganizerID NUMBER NOT NULL,
    TimeSlotID NUMBER NOT NULL,
    Title NVARCHAR2(200) NOT NULL,
    Description CLOB,
    Status NVARCHAR2(50),
    Capacity NUMBER,

    CONSTRAINT FK_Events_Organizers FOREIGN KEY (OrganizerID)
        REFERENCES Organizers(OrganizerID),

-- =========================
-- EVENT REGISTRATIONS
-- =========================
CREATE TABLE EventRegistrCancel ( 
    ID NUMBER, 
    EventID NUMBER NOT NULL, 
    AttendeeID NUMBER NOT NULL, 
    CreatedAt DATE DEFAULT SYSDATE, 
    AttendeeStatus NVARCHAR2(20), 

    CONSTRAINT PK_EventRegistrCancel PRIMARY KEY (ID), 
    CONSTRAINT FK_RegCancel_Events FOREIGN KEY (EventID) 
        REFERENCES Events(EventID), 
    CONSTRAINT FK_RegCancel_Attendees FOREIGN KEY (AttendeeID) 
        REFERENCES Attendees(AttendeeID) 
);

CREATE SEQUENCE seq_eventreg START WITH 1;

CREATE OR REPLACE TRIGGER trg_eventreg
BEFORE INSERT ON EventRegistrCancel
FOR EACH ROW
WHEN (NEW.ID IS NULL)
BEGIN
    :NEW.ID := seq_eventreg.NEXTVAL;
END;
/
